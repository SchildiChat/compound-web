import React from "react";
import { Meta, Title, IconGallery, IconItem } from "@storybook/blocks";

import * as icons from "@vector-im/compound-design-tokens/assets/web/icons";

<Meta title="Icon/Gallery" />

# Icon Gallery

<IconGallery>
  {Object.entries(icons).map(([name, Icon]) => (
    <IconItem
      key={name}
      name={name
        .replace(/Icon$/, "")
        .split(/\.?(?=[A-Z])/)
        .join("-")
        .toLowerCase()}
    >
      <Icon />
    </IconItem>
  ))}
</IconGallery>
